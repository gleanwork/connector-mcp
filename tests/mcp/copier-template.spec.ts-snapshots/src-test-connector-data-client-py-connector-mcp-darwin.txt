from __future__ import annotations

import json
from collections.abc import Sequence
from pathlib import Path
from typing import Any

from glean.indexing import BaseDataClient

from test_connector.models import SourceDocument

_MOCK_DATA_PATH = Path(__file__).parent / "mock_data.json"


class DataClient(BaseDataClient[SourceDocument]):
    """Fetches documents from the external data source.

    Replace the mock implementation with real API calls.
    """

    def get_source_data(self, **kwargs: Any) -> Sequence[SourceDocument]:
        """Return all documents from the data source."""
        with _MOCK_DATA_PATH.open() as f:
            data: list[SourceDocument] = json.load(f)
        return data
